    Linux常用命令–初级 1.pwd #查看当前目录路径 print working directory 2.cd #切换目录change directory cd 直接回到家目录 cd /root 回到家目录 cd ~ 回到家目录 cd - 回到上一次所在目录 cd .. 表示上一级目录 cd . 表示当前目录 3.ls #显示文件 查看文件或者目录是否存在 list -l #显示详细信息 ls -l 显示文件或目录的详细信息 ll 相当有执行了ls -l 命令 ll是别名 ll -i 可以显示iNode号 -a # 显示所有文件及目录包含隐藏的文件 4.touch #创建普通文件 如果文件存在则只修改文件的时间 语法结构： touch 普通文件 # 在当前位置创建普通文件 touch file file2 file3 # 一次性创建多个文件 touch /etc/test.txt # 在指定的目录下创建test.txt 5.mkdir # 创建目录 -p 递归创建 语法结构： mkdir 目录名称 # 创建单个目录 mkdir dir1 dir2 # 创建多个目录 mkdir /opt/目录名称 # 在绝对路径下创建目录 # -p 递归创建 6.tree # 查看目录结构 7.cat # 查看文件的内容 语法结构： cat 文件 cat /目录/文件 8.clear # 清屏 9.mount # 挂载光驱 mount /dev/cdrom /mnt # 把光驱挂载到mnt目录 10.umonut # 卸载挂载点 加-f可以强制卸载 umount /mnt 11.cp # 复制文件 copy 语法结构： cp 源文件 目标位置 cp 源文件 新文件名 cp file1 file2 目录/ # 复制多个文件到目录 cp 目录 目标位置 -r # 复制目录，递归拷贝 12.mv 移动 move 语法结构： mv 源文件 目标位置 mv file1 file2 目标文件 mv 源文件 新文件名 ##### 使用 mv 代替 rm 13.rm # 删除文件或目录 remove 语法结构： rm 文件 rm file1 file2 rm 目录 -f # 强制删除不提示，r 递归删除 14.echo # 输出信息到屏幕或文件中 ，测试使用 语法格式： echo 字符串 # 输出到屏幕 echo 'sdfad' > test.txt # 将输出信息写入文件 echo {1..100} # 输出序列 > 和 >> 只接收正确的 2> 和 2>> 只接收错误的 15.vim 快捷键 视图模式： :set nu # 显示行号 G: 快速移动光标到末尾 10G 移动到第10行 1G 快速移动到第一行 gg: 快速移动光标到第一行 $: 快速到行尾 End 0： 快速到行首 ^ Home / 搜索内容 n 往下搜索 N 往上搜索 删除内容： dd 快速剪切光标所在的行 3dd 快速剪切往下3行 d+End 快速删除光标所在到尾行 d+0 d+Home d+^ 快速删除光标所在行到行首 d+w 删除光标后的单个单词 D 删除光标所在到行尾 u 撤销操作 d+G 快速删除光标所在到文件末尾 p 粘贴剪切复制的内容 3p 快速粘贴3行内容 yy 复制 进入编辑模式：i a o 在光标下一行进入编辑模式 O 在光标上一行进入编辑模式 C 删除光标所在到行尾并进入编辑模式 底行模式： :wq 保存并退出 :q! 强制退出不保存 :set nu 显示行号 :set nonu 取消显示行号 :set paste 粘贴外部代码时避免自动缩进错乱。 视图模式下：ZZ 保存并退出 替换： :s/// 只替换光标所在行的第一个单纯 :s///g 替换光标所在整行的单词 :%s///g 替换文件中的所有单词 ctrl+v # 块模式，批量选中， 输入I （shift+i）进入到编辑模式，数据内容，按2次esc shift+v # 批量选行 16.hostnamectl set-hostname xxx # 永久修改主机名 hostname xxx # 临时修改主机名 cat /etc/hostname # 查看主机名 17. lscpu # 查看cpu信息 18. free -h # 查看内存的信息 19. df -h # 查看磁盘信息 20. uptime # 查看系统负载 21. w # 查看系统负载和用户登录信息 22. tail # 默认显示文件末尾10行 -n数据 # 显示n行，简写可以用 tail -3 文件 -f # 不退出，实时显示文件的变化 tailf == tail -f # tailf不是别名是独立的命令 -F # 了解 实时显示文件的变化，文件不存在也可以 23. hand # 默认显示文件头10行 -n 参数 24. grep # 过滤文件中的单词，模糊过滤 25. less # 一页一页查看文件的内容，查看大文件比较好用 --- 空格或f向下翻页 26. more 了解 27. wc # wc -l 统计行数，或者统计输出到屏幕的行数 || cat file|wc -l 28. sort # 排序，默认按照第一列进行正序排序 sort -r # 逆序排序 sort -n # 按数字正序排序 sort -nr # 按数字的逆序排序 sort -nk2 # 对第二列进行数字正序排序 +r 逆序（-nrk2） 29. uniq # 去重（把挨着的相同内容进行去重） uniq -c # 去重统计 30. du -sh # 查看文件大小 31. stat # 查看文件详细信息，包含文件时间 find # 查找文件 1.find ./ -name "" # 按名称查找 2.find ./ -iname "" # 不区分大小写 3.find ./ -inum "" # 按iNode号码查找 4.find ./ -type f/d/c. # 按文件类型查找 5.find ./ -type f -name "" # 联合查询默认并且 同时成立 6.find ./ -maxdepth 1 -type f # 按照深度等级查找 7.find ./ -size 10M -10M +10M # 按照文件大小查找 8.find ./ -size 10M -size +15M # 并且的意思，需要符合所给的条件 9.find ./ -size 10M or -name "1.txt" # 查找文件名为1.txt 或者是10M的文件 10.find ./ -mtime +7 -7 # 按照文件修改时间查找 find将结果交给其他命令 xargs、exec、`` find ./ -name "1.txt" | xargs rm -r # 删除目录需要加-r find ./ -name "1.txt" | xargs ls -l # 查看详细信息 find ./ -name "1.txt" | xargs -i cp {} /opt/ # 复制 find ./ -name "1.txt" -exec rm {} \; find ./ -name "1.txt" -exec ls -l {} \; find ./ -name "1.txt" -exec cp {} /opt \; rm -rf `find ./ -name "1.txt"` ll `find ./ -name "1.txt"` cp `find ./ -name "1.txt"` /opt/ 三个特殊符号：$$ || ; tar # 解压缩文件 tar zcvf all.tar.gz 文件1 文件2 文件3 （尽量进入相对路径打包） z：使用zip压缩 c：create 创建 v：verbose显示压缩过程，可以省略此参数 f：file文件 xf：解压缩 # 默认解压到当前目录 -C：指定解压的目录 # tar xf xxx.tar.gz -C /tmp/ tf：查看压缩包里的文件 扩展： --exclude=1.txt # 排除文件 --exclude-from=file # 排除file中多个文件 zip 压缩 1.zip all.zip *.txt # 打包 2.unzip all.zip # 解压到当前目录 3.unzip all.zip -d /tmp # 指定解压的位置 Linux 的配置文件 1. /etc/sysconfig/network-scripts/ifcfg-eth0 # 网卡配置文件 2. /etc/hosts # 本地DNS域名解析服务 3. /etc/hostname # 主机名称的配置文件 里面存放的主机名 、、、hostnamectl set-hostname zhujiming (临时加永久生效) 4. /etc/resolv.conf # 作用：存放DN配置文件的，DNS配置文件， 配置后生效 5. /etc/fstab # 作用：开机自动挂载 UUID=04aaab6c-3fcf-4dac-8265-034fe807452e / xfs defaults 0 0 第一列：设备的名称 可以是UUID 可以使用设备的名称 第二列：挂载点，设备的入口，类似通过/mnt进入cdrom 第三列：文件系统类型 第四列：default挂载参数，默认即可 第五列：0 不自检 开机是否自检 1自检 第六列：0 不备份 开机是否备份 1备份 6. /etc/rc.local # 作用：开机自动执行文件中的命令 -- 需要有执行权限 ——chmod +x /etc/rc.local 7. /etc/inittab # 文件（了解） 运行级别 0 # 关机 1 # 单用户模式，忘记密码可以进入1级别 2 # 表示多用户，但是不支持NFS，网络文件系统（缺少很多功能） 3 # 表示完全多用户 我们默认所在的界面 4 # 保留待开发 5 # 表示桌面运行级别 需要提前安装DESKTOP包 6 # 表示重启 8. /etc/motd # 开机显示文件内容 9. /var/log/mesages # 系统日志文件 10. /var/log/secure # 系统登录和退出日志 Linux快捷键 1.ctrl+c # 终止进程 2.ctrl+l # 清屏 3.ctrl+a # 快速移动光标到行首 Home 4.catrl+e # 快速移动光标到行尾 End 5.catrl+方向键 # 移动一个单词 6.catrl+u # 剪切光标到行首 7.catrl+y # 粘贴
