Linux常用命令–初级#

1.pwd	#查看当前目录路径 print working directory
2.cd	#切换目录change directory
	cd		    直接回到家目录
	cd /root	回到家目录
	cd ~		回到家目录
	cd -		回到上一次所在目录
	cd ..		表示上一级目录
	cd .		表示当前目录
3.ls	#显示文件 查看文件或者目录是否存在 list
	-l	#显示详细信息
		ls -l 显示文件或目录的详细信息
		ll 相当有执行了ls -l 命令 ll是别名
	ll -i 可以显示iNode号
	-a	# 显示所有文件及目录包含隐藏的文件
4.touch	#创建普通文件	如果文件存在则只修改文件的时间
语法结构：
		touch 普通文件				# 在当前位置创建普通文件
		touch file file2 file3		# 一次性创建多个文件
		touch /etc/test.txt		# 在指定的目录下创建test.txt
5.mkdir	# 创建目录 -p 递归创建
语法结构：
		mkdir 目录名称		# 创建单个目录
		mkdir dir1 dir2		# 创建多个目录
		mkdir /opt/目录名称		# 在绝对路径下创建目录
		# -p 递归创建
6.tree	# 查看目录结构
7.cat	# 查看文件的内容
语法结构：
		cat 文件
		cat /目录/文件
8.clear	# 清屏
9.mount # 挂载光驱
	mount /dev/cdrom /mnt	# 把光驱挂载到mnt目录
10.umonut	# 卸载挂载点 加-f可以强制卸载
	umount	/mnt
11.cp	# 复制文件 copy
语法结构：
		cp 源文件 目标位置
		cp 源文件 新文件名
		cp file1 file2 目录/	# 复制多个文件到目录
		cp 目录 目标位置
		-r # 复制目录，递归拷贝
12.mv	移动 move 
语法结构：
		mv 源文件 目标位置
		mv file1 file2 目标文件
		mv 源文件 新文件名
##### 使用 mv 代替 rm
13.rm	# 删除文件或目录 remove
语法结构：
		rm 文件
		rm file1 file2
		rm 目录
		-f # 强制删除不提示，r 递归删除
		
14.echo	# 输出信息到屏幕或文件中 ，测试使用
语法格式：
		echo 字符串	# 输出到屏幕
		echo 'sdfad' > test.txt	# 将输出信息写入文件
		echo {1..100}	# 输出序列
> 和 >> 只接收正确的
2> 和 2>> 只接收错误的

31.	stat	# 查看文件详细信息，包含文件时间

	find   # 查找文件
        1.find ./ -name ""	# 按名称查找
        2.find ./ -iname ""	# 不区分大小写
        3.find ./ -inum ""	# 按iNode号码查找
        4.find ./ -type f/d/c.	# 按文件类型查找
        5.find ./ -type f -name ""	# 联合查询默认并且 同时成立
        6.find ./ -maxdepth 1 -type f # 按照深度等级查找
        7.find ./ -size 10M -10M +10M	# 按照文件大小查找
        8.find ./ -size 10M -size +15M	# 并且的意思，需要符合所给的条件
        9.find ./ -size 10M or -name "1.txt" # 查找文件名为1.txt 或者是10M的文件
        10.find ./ -mtime +7 -7	# 按照文件修改时间查找
     find将结果交给其他命令 xargs、exec、``
     	find ./ -name "1.txt" | xargs rm -r	# 删除目录需要加-r
     	find ./ -name "1.txt" | xargs ls -l	# 查看详细信息
     	find ./ -name "1.txt" | xargs -i cp {} /opt/	# 复制
     	
     	find ./ -name "1.txt" -exec rm {} \;
     	find ./ -name "1.txt" -exec ls -l {} \;
     	find ./ -name "1.txt" -exec cp {} /opt \;
     	
     	rm -rf `find ./ -name "1.txt"`
     	ll `find ./ -name "1.txt"`
     	cp `find ./ -name "1.txt"` /opt/
三个特殊符号：$$ || ;
	
    tar	# 解压缩文件 tar zcvf all.tar.gz 文件1 文件2 文件3 （尽量进入相对路径打包）

    	扩展：

    zip	压缩
    	1.zip all.zip *.txt	# 打包

    	3.unzip all.zip -d /tmp	# 指定解压的位置

Linux 的配置文件#

1.	/etc/sysconfig/network-scripts/ifcfg-eth0	# 网卡配置文件
2.	/etc/hosts	# 本地DNS域名解析服务
3.	/etc/hostname	# 主机名称的配置文件  里面存放的主机名 、、、hostnamectl set-hostname zhujiming (临时加永久生效)
4.	/etc/resolv.conf	# 作用：存放DN配置文件的，DNS配置文件， 配置后生效
5.	/etc/fstab		# 作用：开机自动挂载
	UUID=04aaab6c-3fcf-4dac-8265-034fe807452e /                       xfs     defaults        0 0
	第一列：设备的名称 可以是UUID 可以使用设备的名称
	第二列：挂载点，设备的入口，类似通过/mnt进入cdrom
	第三列：文件系统类型
	第四列：default挂载参数，默认即可
	第五列：0 不自检 开机是否自检 1自检
	第六列：0 不备份 开机是否备份 1备份
6. /etc/rc.local	# 作用：开机自动执行文件中的命令 -- 需要有执行权限 ——chmod +x /etc/rc.local
7. /etc/inittab		# 文件（了解） 运行级别    
	0	# 关机	
	1	# 单用户模式，忘记密码可以进入1级别	
	2	# 表示多用户，但是不支持NFS，网络文件系统（缺少很多功能）
	3	# 表示完全多用户 我们默认所在的界面
	4	# 保留待开发
	5 	# 表示桌面运行级别  需要提前安装DESKTOP包
	6	# 表示重启
8. /etc/motd	# 开机显示文件内容
9. /var/log/mesages	# 系统日志文件
10.	/var/log/secure	# 系统登录和退出日志
